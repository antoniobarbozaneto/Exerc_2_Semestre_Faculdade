			int[] matrizB = new int[10];

			for (int i = 0; i < 10; i++)

				matrizB[i] = (matrizA[i]*-1);
				Console.WriteLine("Matriz A: POSIÇÃO: "+i+" Valor: "+matrizA[i]);

			}
			

			Console.ReadKey();
		}
	}
